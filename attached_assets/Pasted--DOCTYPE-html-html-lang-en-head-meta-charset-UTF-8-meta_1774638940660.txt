// Menu highlight slider
(function initMenuHighlights(){
  const track=document.getElementById('menuHighlights');
  const prevBtn=document.getElementById('msPrev');
  const nextBtn=document.getElementById('msNext');
  const dotsWrap=document.getElementById('msDots');
  if(!track || !prevBtn || !nextBtn || !dotsWrap) return;

  const items=Array.from(track.querySelectorAll('.ms-item'));
  if(items.length===0) return;

  const prefersReduced=window.matchMedia && window.matchMedia('(prefers-reduced-motion: reduce)').matches;
  let activeIndex=0;
  let rafId=null;
  let autoplayId=null;
  let paused=false;

  function scrollToIndex(index){
    const i=(index+items.length)%items.length;
    const item=items[i];
    const left=item.offsetLeft-(track.clientWidth-item.clientWidth)/2;
    track.scrollTo({left,behavior:prefersReduced?'auto':'smooth'});
  }

  function setActive(index){
    activeIndex=index;
    items.forEach((el,i)=>el.classList.toggle('is-active',i===activeIndex));
    Array.from(dotsWrap.children).forEach((d,i)=>d.setAttribute('aria-current',i===activeIndex?'true':'false'));
  }

  function computeActive(){
    const center=track.getBoundingClientRect().left + track.clientWidth/2;
    let best=0;
    let bestDist=Infinity;
    items.forEach((it,i)=>{
      const r=it.getBoundingClientRect();
      const c=r.left + r.width/2;
      const dist=Math.abs(c-center);
      if(dist<bestDist){bestDist=dist;best=i;}
    });
    setActive(best);
  }

  function onScroll(){
    if(rafId) return;
    rafId=requestAnimationFrame(()=>{
      rafId=null;
      computeActive();
    });
  }

  // Build dots
  items.forEach((_,i)=>{
    const b=document.createElement('button');
    b.type='button';
    b.className='ms-dot';
    b.setAttribute('aria-label',`Go to highlight ${i+1}`);
    b.setAttribute('aria-current',i===0?'true':'false');
    b.addEventListener('click',()=>scrollToIndex(i));
    dotsWrap.appendChild(b);
  });

  prevBtn.addEventListener('click',()=>scrollToIndex(activeIndex-1));
  nextBtn.addEventListener('click',()=>scrollToIndex(activeIndex+1));
  track.addEventListener('scroll',onScroll,{passive:true});
  track.addEventListener('keydown',(e)=>{
    if(e.key==='ArrowLeft'){e.preventDefault();scrollToIndex(activeIndex-1);}
    if(e.key==='ArrowRight'){e.preventDefault();scrollToIndex(activeIndex+1);}
  });

  function pause(){paused=true;}
  function resume(){paused=false;}
  track.addEventListener('mouseenter',pause);
  track.addEventListener('mouseleave',resume);
  track.addEventListener('focusin',pause);
  track.addEventListener('focusout',resume);

  function startAutoplay(){
    if(prefersReduced) return;
    autoplayId=setInterval(()=>{
      if(paused) return;
      scrollToIndex(activeIndex+1);
    },4500);
  }

  computeActive();
  startAutoplay();
  window.addEventListener('resize',()=>computeActive(),{passive:true});
})();
